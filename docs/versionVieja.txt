            this.errorMessage = 'Por favor, complete todos los campos del formulario.';
            this.showErrorModal = true;
            return;

            this.stockPorUP[this.selectedUpId] = [];

            s.especie_id == this.form.especie_id &&

            s.raza_id == this.form.raza_id
        );

            if (this.registrationType === 'altas') {
                stockItem.cantidad = parseInt(stockItem.cantidad) + parseInt(this.form.cantidad);

                    this.showErrorModal = true;
                    return; // Stop if stock would go negative

                    if (itemIndex > -1) {
                        this.stockPorUP[this.selectedUpId].splice(itemIndex, 1);
                    }
                } else {
                    // Otherwise, just update the quantity
                    stockItem.cantidad = newCantidad;
                }
            }
        } else { // If stock item does not exist
            if (this.registrationType === 'bajas') {
                this.errorMessage = 'No puede registrar una baja para un tipo de animal que no tiene en stock.';

            categoria_nombre: categoria.nombre,
            raza_nombre: raza.nombre,
            tipo: this.registrationType
        };

        this.showModal = false;
        this.form = { especie_id: '', categoria_id: '', raza_id: '', cantidad: 1 };
    },
    cancelChanges() {
        this.stockPorUP = JSON.parse(JSON.stringify(this.initialStock));
        Livewire.dispatch('limpiar-resumen');
        this.showCancelModal = false;
    }
